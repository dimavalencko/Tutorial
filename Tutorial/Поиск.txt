Для поиска необходимо создать свой приватный метод и назвать его как-нибудь(У меня это UpdateCloth).
В нем мы создаем переменную которая содержить данные нужной таблицы по которой будет идти поиск.
дальше мы этой переменной присваиваем её же, где Артикул_Ткани будет равен тексту введенному в TBSearch.
Т.е он будет смотреть на данные введённые в TBSearch и искать в указанной колонке данные которые совпадают свведёными, иначе он просто убирает лишние данные из DataGrid по которому идет поиск.

private void UpdateCloth()
        {
            var currentCloth = FactoryEntities.GetContext().Склад_ткани.ToList();//Тут надо поменять название фраемворка и таблицы в которой надо искать данные

            currentCloth = currentCloth.Where(p => p.Артикул_ткани.ToString().Contains(TBSearch.Text.ToString())).ToList();//Здесь надо поменять Артикул_Ткани на свое название колонки по которой нужно искать и надо поменять название TBSearch на свое, куда вводится значение для поиска.
            DGridOstatok.ItemsSource = currentCloth.ToList();//Тут меняем название DGridOstatok на название своего грида.
        }


После этого мы создаем у ТекстБокса в который вводятся данные для поиска событие TextChanged.В него мы помещаем Название метода который нужно вызвать.В данном примере это UpdateCloth();.

        private void TBSearch_TextChanged(object sender, TextChangedEventArgs e)
        {
            UpdateCloth();
        }

И также в данной части кода тоже нужно проописать название метода

public OstatkiTkani()
        {
            InitializeComponent();//Где написана вот этафраза туда и помещать.Это главный метод формы.
            UpdateCloth();
        }
